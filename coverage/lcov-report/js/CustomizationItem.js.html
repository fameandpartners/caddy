import React from 'react';
import autoBind from 'react-autobind';
 
export default class CustomizationItem extends React.Component
{
    constructor( props )
    {
        super( props );
        autoBind( this );
    }
 
    updateValue()
    {
        this.props.update( parseInt(this.props.customizationKey), this.textInput.value );
    }
    
    render()
    {
        return(
            <li>
              Customization Name: <input type="text" onKeyUp={this.updateValue } ref={(input) => { this.textInput = input;  }} />  <button onClick={() => this.props.deleteCustomization( this.props.customizationKey ) }>Remove</button>
            </li>
        );
    }
}
